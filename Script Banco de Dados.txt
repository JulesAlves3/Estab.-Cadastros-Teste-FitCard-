CREATE TABLE [dbo].[CATEGORIA_ESTAB] (
    [Id_CATEGORIA] INT           IDENTITY (1, 1) NOT NULL,
    [CATEGORIAS]   VARCHAR (100) NOT NULL,
    PRIMARY KEY CLUSTERED ([Id_CATEGORIA] ASC)
);

CREATE TABLE [dbo].[DADOS_ESTAB] (
    [Id_ESTAB]     INT           IDENTITY (1, 1) NOT NULL,
    [RAZAO]        VARCHAR (100) NOT NULL,
    [FANTASIA]     VARCHAR (100) NULL,
    [CNPJ]         VARCHAR (100) NOT NULL,
    [EMAIL]        VARCHAR (100) NULL,
    [ENDERECO]     VARCHAR (100) NULL,
    [CIDADE]       VARCHAR (100) NULL,
    [TELEFONE]     VARCHAR (100) NULL,
    [AGENCIACONTA] VARCHAR (100) NULL,
    [Id_ESTADO]    INT           NOT NULL,
    [Id_CATEGORIA] INT           NOT NULL,
    [Id_STATUS]    INT           NOT NULL,
    [DATA]         VARCHAR (100) NULL,
    PRIMARY KEY CLUSTERED ([Id_ESTAB] ASC),
    CONSTRAINT [FK_DADOS_ESTAB_CATEGORIA_ESTAB] FOREIGN KEY ([Id_CATEGORIA]) REFERENCES [dbo].[CATEGORIA_ESTAB] ([Id_CATEGORIA]),
    CONSTRAINT [FK_DADOS_ESTAB_ESTADO_ESTAB] FOREIGN KEY ([Id_ESTADO]) REFERENCES [dbo].[ESTADO_ESTAB] ([Id_ESTADO]),
    CONSTRAINT [FK_DADOS_ESTAB_STATUS_ESTAB] FOREIGN KEY ([Id_STATUS]) REFERENCES [dbo].[STATUS_ESTAB] ([Id_STATUS])
);

CREATE TABLE [dbo].[ESTADO_ESTAB] (
    [Id_ESTADO] INT           IDENTITY (1, 1) NOT NULL,
    [ESTADO]    VARCHAR (100) NOT NULL,
    PRIMARY KEY CLUSTERED ([Id_ESTADO] ASC)
);

CREATE TABLE [dbo].[STATUS_ESTAB] (
    [Id_STATUS] INT           IDENTITY (1, 1) NOT NULL,
    [STATUS]    VARCHAR (100) NOT NULL,
    PRIMARY KEY CLUSTERED ([Id_STATUS] ASC)
);
